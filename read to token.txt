import pandas as pd
import hashlib

def generate_token(data):
    """
    Generate a token using a hash function.
    :param data: The input data to generate the token from.
    :return: A unique token as a string.
    """
    return hashlib.sha256(data.encode()).hexdigest()

def read_excel_and_create_tokens(file_path):
    """
    Read an Excel file and create tokens for each row.
    :param file_path: Path to the Excel file.
    """
    try:
        # Read the Excel file
        df = pd.read_excel(file_path)

        # Iterate through rows and generate tokens
        tokens = {}
        for index, row in df.iterrows():
            row_data = str(row.to_dict())  # Convert row to a string representation
            token = generate_token(row_data)
            tokens[index] = token

        # Print or save the tokens
        for index, token in tokens.items():
            print(f"Row {index}: Token = {token}")

    except Exception as e:
        print(f"Error: {e}")
